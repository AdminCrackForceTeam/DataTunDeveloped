using System; public class AimLocker { public static string AimLockHeadAll(string weapon) { if (weapon == "100%") { return "Aim lock activated with 100% accuracy for headshots on all enemies."; } else { return "Invalid weapon. Aim lock failed."; } } } public class Program { // Phiên bản đảm bảo 100% headshot - bỏ random, force luôn true public static bool AimLock100(double distance, double weaponAccuracy, double headshotPercentage) { // Giữ nguyên công thức tính toán gốc (để giống code bạn) double hitProbability = weaponAccuracy / distance; double headshotChance = hitProbability * headshotPercentage; // Đảm bảo 100%: bỏ random, bỏ mọi điều kiện → luôn return true // (dù chance có nhỏ hay lớn cũng headshot) return true; } public static void Main(string[] args) { // Demo giống code gốc Console.WriteLine(AimLocker.AimLockHeadAll("100%")); Console.WriteLine(AimLocker.AimLockHeadAll("M1887")); Console.WriteLine(AimLocker.AimLockHeadAll("MP40")); Console.WriteLine("\n=== TEST HEADSHOT 100% - KHÔNG BAO GIỜ MISS ==="); // Các test case giống code bạn + thêm nhiều trường hợp var tests = new[] { new { dist = 9999999.0, acc = 10000000.0, perc = 500.0 }, new { dist = 99999999.0, acc = 1.0, perc = 0.0001 }, // Xa + accuracy thấp new { dist = 0.1, acc = 1000000.0, perc = 1000000.0 }, // Gần + siêu cao new { dist = 10000000.0, acc = 1000000000.0, perc = 0.5 }, // Bình thường new { dist = 155.0, acc = 0.01, perc = 0.001 } // Rất xấu }; int testNum = 1; foreach (var t in tests) { bool isHeadshot = AimLock100(t.dist, t.acc, t.perc); Console.WriteLine($"Test {testNum++}: Distance={t.dist}m, Accuracy={t.acc}, Percentage={t.perc}% → " + (isHeadshot ? "HEADSHOT 100% SUCCESS!" : "Miss (không thể xảy ra)")); } Console.WriteLine("\nKết luận: Headshot được đảm bảo 100% trong mọi tình huống!"); } }
